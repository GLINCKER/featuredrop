# Database Adapters

Use server adapters when dismissal/read state must persist across devices.

Import adapters from `featuredrop/adapters`.

<Callout type="info">
  `RemoteAdapter` runs in the browser and talks to your API. Database adapters run on the server behind that API.
</Callout>

## Remote contract

Your API usually supports:

- `GET /api/featuredrop/state` -> `{ watermark, dismissedIds }`
- `POST /api/featuredrop/dismiss` -> `{ featureId }`
- `POST /api/featuredrop/dismiss-all` -> `{ watermark }`

```ts filename="app/api/featuredrop/state/route.ts"
import { NextResponse } from 'next/server'
import { PostgresAdapter } from 'featuredrop/adapters'
import { Pool } from 'pg'

const pool = new Pool({ connectionString: process.env.DATABASE_URL })

function getAdapter(userId: string) {
  return new PostgresAdapter({
    userId,
    query: (sql, params) => pool.query(sql, params as unknown[])
  })
}

export async function GET() {
  const userId = 'current-user-id'
  const adapter = getAdapter(userId)
  await adapter.sync()
  return NextResponse.json({
    watermark: adapter.getWatermark(),
    dismissedIds: Array.from(adapter.getDismissedIds())
  })
}
```

```ts filename="app/api/featuredrop/dismiss/route.ts"
import { NextResponse } from 'next/server'
import { MongoAdapter } from 'featuredrop/adapters'
import { MongoClient } from 'mongodb'

const client = new MongoClient(process.env.MONGODB_URI!)

function getAdapter(userId: string) {
  const collection = client.db('production').collection('featuredrop_state')
  return new MongoAdapter({ userId, collection })
}

export async function POST(req: Request) {
  const { featureId } = await req.json()
  const userId = 'current-user-id'
  const adapter = getAdapter(userId)
  adapter.dismiss(featureId)
  return NextResponse.json({ ok: true })
}
```

```ts filename="app/api/featuredrop/dismiss-all/route.ts"
import { NextResponse } from 'next/server'
import { SupabaseAdapter } from 'featuredrop/adapters'
import { createClient } from '@supabase/supabase-js'

const supabase = createClient(
  process.env.NEXT_PUBLIC_SUPABASE_URL!,
  process.env.SUPABASE_SERVICE_ROLE_KEY!
)

export async function POST(req: Request) {
  const { watermark } = await req.json()
  const userId = 'current-user-id'

  const adapter = new SupabaseAdapter({
    userId,
    client: supabase,
    tableName: 'featuredrop_state',
    realtime: true
  })

  await adapter.dismissAll(new Date(watermark))
  return NextResponse.json({ ok: true })
}
```

```ts filename="app/api/featuredrop/state/route.ts"
import { NextResponse } from 'next/server'
import { RedisAdapter } from 'featuredrop/adapters'
import Redis from 'ioredis'

const client = new Redis(process.env.REDIS_URL!)

export async function GET() {
  const userId = 'current-user-id'
  const adapter = new RedisAdapter({ userId, client, keyPrefix: 'fd:' })
  await adapter.sync()
  return NextResponse.json({
    watermark: adapter.getWatermark(),
    dismissedIds: Array.from(adapter.getDismissedIds())
  })
}
```

## Adapter contract

All DB adapters implement `ServerStorageAdapter`, so you can swap backends without changing React integration.
